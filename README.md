# MPV播放器配置文件

```
# 高优先级
priority=high

# GPU
profile=gpu-hq
gpu-api=d3d11
hwdec=auto

# 视频输出格式
d3d11-output-format=auto

# RGB范围
video-output-levels=full

# 声音
volume=30
volume-max=100

# 窗口缩放
geometry=55%

# 缓存
cache=yes
cache-default=4000000

# 字幕
slang=zh,chi,chs,sc,zho,en,eng
sub-codepage=enca:zh:utf8

# UI
no-border
ontop
focus-on-open
snap-window=yes
```
